Usage

Include script within <head> or before </body>

<script src="msa_widget.js"></script>

Create a new widget instance

<script>

var app = new msa_widget();
.
.

Optionally change some messages...

// appears when you mouse over the widget strip
app.messages.widget_tooltip = 'Get My StarHub app!';

// the label on the widget strip
app.messages.widget_headline = '...';

// when widget is expanded, this is the title below the image, for non Android or iOS browsers
app.messages.widget_cta_headline_universal = '....';

// when widget is expanded, this is the title below the image, for iOS browsers
app.messages.widget_cta_headline_ios = '....';

// when widget is expanded, this is the title below the image, for Android browsers
app.messages.widget_cta_headline_android = '....';

// when widget is expanded, this is the description
app.messages.widget_cta_description = '....';

Or change some settings

// the MSA Google Play link
app.settings.android_link = '...';

// the MSA app store link
app.settings.ios_link = '...';

// the MSA Google Play store deeplink
app.settings.android_store_link = '...';

// the MSA app store deeplink
app.settings.ios_store_link = '...';

// boolean. use the store deeplinks? defaults to true. This also requires that user is using iOS or Android browser.
app.settings.use_store_link = true or false;

// the MSA Youtube video URL. Video will play when widget is displayed	
app.settings.youtube_video = '...';

// the MSA key visual displayed when the widget is opened
app.settings.msa_image = '...';

Finally, run it

.
.
app.run();

</script>
